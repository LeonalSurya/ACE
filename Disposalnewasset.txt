        function SelectRow(CurrentRow, RowIndex) {

        function DiscountCalculation() {
            var Qty;
            var Price;
            var DiscountPer;
            var Discount;

            if (isNaN(document.getElementById('ctl00_ContentPlaceHolder1_txtQty').value) || document.getElementById('ctl00_ContentPlaceHolder1_txtQty').value == '') {
                Qty = 0;
            }
            else {
                Qty = parseFloat(document.getElementById('ctl00_ContentPlaceHolder1_txtQty').value);
            }

            if (isNaN(document.getElementById('ctl00_ContentPlaceHolder1_txtItemPrice').value) || document.getElementById('ctl00_ContentPlaceHolder1_txtItemPrice').value == '') {
                Price = 0;
            }
            else {
                Price = parseFloat(document.getElementById('ctl00_ContentPlaceHolder1_txtItemPrice').value);
            }

            if (isNaN(document.getElementById('ctl00_ContentPlaceHolder1_txtDiscountPer').value) || document.getElementById('ctl00_ContentPlaceHolder1_txtDiscountPer').value == '') {
                DiscountPer = 0;
                Discount = 0;
            }                            
            else {
                DiscountPer = parseFloat(document.getElementById('ctl00_ContentPlaceHolder1_txtDiscountPer').value);                
                Discount = (Qty * Price) * (DiscountPer / 100);
            }            
            if (Discount == 0) {            
                document.getElementById('ctl00_ContentPlaceHolder1_txtItemDiscount').value = '';
            }
            else {
                document.getElementById('ctl00_ContentPlaceHolder1_txtItemDiscount').value = Discount;
            }
            
        }        

        function Calculation() 
        {            
            var Qty;
            var Price;
            var Discount;
            var GrossTotal;
            var Tax;            
            var Total;
            
            if (isNaN(document.getElementById('ctl00_ContentPlaceHolder1_txtQty').value) || document.getElementById('ctl00_ContentPlaceHolder1_txtQty').value == '') 
            {
                Qty = 0;
            }
            else 
            {
                Qty = parseFloat(document.getElementById('ctl00_ContentPlaceHolder1_txtQty').value);
            }

            if (isNaN(document.getElementById('ctl00_ContentPlaceHolder1_txtItemPrice').value) || document.getElementById('ctl00_ContentPlaceHolder1_txtItemPrice').value == '') 
            {
                Price = 0;
            }
            else 
            {
                Price = parseFloat(document.getElementById('ctl00_ContentPlaceHolder1_txtItemPrice').value);
            }

            //CFact = document.getElementById('ctl00_ContentPlaceHolder1_ddlUnits')[document.getElementById('ctl00_ContentPlaceHolder1_ddlUnits').selectedIndex].value;
            
            if (isNaN(document.getElementById('ctl00_ContentPlaceHolder1_txtDiscountPer').value) || document.getElementById('ctl00_ContentPlaceHolder1_txtDiscountPer').value == '') 
            {
                if (isNaN(document.getElementById('ctl00_ContentPlaceHolder1_txtItemDiscount').value) || document.getElementById('ctl00_ContentPlaceHolder1_txtItemDiscount').value == '') 
                {
                    Discount = 0;
                }
                else
                {
                    Discount = parseFloat(document.getElementById('ctl00_ContentPlaceHolder1_txtItemDiscount').value);
                }
                GrossTotal = (Qty * Price) - Discount;
            }
            else
            {
                DiscountPer = parseFloat(document.getElementById('ctl00_ContentPlaceHolder1_txtDiscountPer').value);
                document.getElementById('ctl00_ContentPlaceHolder1_txtItemDiscount').value=(Qty * Price) * (DiscountPer / 100);
                GrossTotal = (Qty * Price) - ((Qty * Price) * (DiscountPer / 100));
            }
            document.getElementById('ctl00_ContentPlaceHolder1_txtItemGross').value = GrossTotal;

            if (isNaN(document.getElementById('ctl00_ContentPlaceHolder1_txtItemTax').value) || document.getElementById('ctl00_ContentPlaceHolder1_txtItemTax').value == '') 
            {
                Tax = 0;
            }
            else 
            {
                Tax = parseFloat(document.getElementById('ctl00_ContentPlaceHolder1_txtItemTax').value);
            }            
            Total = GrossTotal + Tax;
            document.getElementById('ctl00_ContentPlaceHolder1_txtTotal').value = Total;            
        }
